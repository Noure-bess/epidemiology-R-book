5 HLIR function 

HLIR_fun <- function(t, y, par) {
  # Variables
  H <- y[1]
  L <- y[2]
  I <- y[3]
  R <- y[4]
  beta <- par$beta
  gama <- par$gama
  mu <- par$mu

  # Right hand side of the model
  dH <- -beta * H * I
  dL <- beta * H * I - gama * L
  dI <- gama * L - mu * I
  dR <- mu * I
  return(list(c(dH, dL, dI, dR)))
}
